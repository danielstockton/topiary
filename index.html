Usage

 
                
[topiary "0.1.0-SNAPSHOT"]
                
              
                
(ns css.main
  (:require [garden.def :refer [defstyles]]
            [garden.units :refer [px em]]
            [garden.color :refer [hsl]]
            [topiary.core :as topiary]))

;; Assoc-in values at any point in the config.
(def conf (-> topiary/config
              (assoc-in [:colors :1] "#333")))

;; Define your own additional styles.
(def styles [[:.right {:float :right}]
             [:#content {:margin-top (em 2)}]
             [:.editable [:input {:margin 0 :padding 0}]]
             [:#files [:li {:margin-left 0
                            :padding-top (em 0.8)
                            :padding-bottom (em 0.8)}
                       [:div.inner {:border "1px solid #333"
                                    :padding (em 1)}
                        [:div.info {:margin-bottom (em 1)
                                    :font-size (em 0.8)}]
                        [:div.buttons [:a {:margin-right (em 1)}]]]]]])

;; Choose only what you need.
(defn all
  [config]
  [(topiary/base config)
   (topiary/screen-small config)
   (topiary/screen-medium config)
   (topiary/screen-large config)
   (topiary/utility config)
   (topiary/body config)
   (topiary/headings config)
   (topiary/nav-top config)
   (topiary/buttons config)
   (topiary/lists config)
   (topiary/forms config)
   (topiary/grid config)
   (topiary/modal config)
   (topiary/tables config])

(def all (conj (all conf) styles))

(defstyles main all)
